93 bool generateMesh(const GeomModel *geom, getfem::mesh *mesh,
94  const ConfigData &meshingConfig) {
95  FEMDKLOGGER().info("Start mesh generation BGD/Getfem");
96  assert(NULL != geom);
97  assert(NULL != mesh);
98 
99  femdk::GMSHMesher mesher;
100 
101  const GeomModelBGD *gmptr = static_cast<const GeomModelBGD*>(geom);
102  bgd::BGD *gp = gmptr->getImplementation();
103  if (NULL == gp) {
104  FEMDK_ERROR("NULL BGD pointer");
105  }
106 
107  mesher.generate(*gp, *mesh);
108  std::cerr << "HERE " << (int)mesh->nb_points() << "\n";
109 
110  FEMDKLOGGER().info("Finish mesh generation BGD/Getfem");
