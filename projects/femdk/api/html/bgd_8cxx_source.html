187  void BGD::export_off_face(std::ostream& out, const std::vector<double> &coords, int dim,
188  double z) {
189  out << "OFF" << "\n";
190  int nvert = coords.size()/dim;
191  int nfaces = 1; //single face
192  int nedges = 6; // not used
193  assert(coords.size() == dim*nvert);
194  out << nvert << " " << nfaces << " " << nedges << "\n";
195  std::vector<double>::const_iterator ci = coords.begin();
196  for (int i=0; i<nvert; i++) {
197  out << *ci << " "; ci++;
198  out << *ci << " "; ci++;
199  if (2 == dim) {
200  out << z << "\n";
201  } else {
202  out << *ci << " "; ci++;
203  }
204  }
205  out << nvert << " ";
206  for (int i=0; i<nvert; i++) {
207  out << i << " ";
208  }
209  out << "\n";
210  }
