58 shape_type_t
59 str2shape_type(const std::string &name, bool casesensitive) {
60  if (casesensitive) {
61  for (int i=0; i<BGD_SHAPE_MAX; i++) {
62  if (name == std::string(bgd_shape_names[i])) {
63  return shape_type_t(i);
64  }
65  }
66  } else {
67  for (int i=0; i<BGD_SHAPE_MAX; i++) {
68  if(boost::iequals(name, bgd_shape_names[i])) {
69  return shape_type_t(i);
70  }
71  }
72  }
73  std::string msg="Invalid BGD shape type: ";
74  msg +=name;
75  throw FEMDKInvalidArgs(msg);
76  return BGD_SHAPE_MAX;
77 }

93 BGD*
94 ShapeFactory::getShape(const std::string &shapeTypeName) const {
95 
96  if (false == checkIfSupported(shapeTypeName)) {
97  std::string msg("Invalid shape name: ");
98  msg += shapeTypeName;
99  throw BGDException(msg);
100  return NULL;
101  }
102 
103  shape_type_t mt = str2shape_type(shapeTypeName, myCaseSensitive);
104 
105  switch (mt) {
106  //case BGD_ANULUS : return new Anulus; break;
107  case BGD_ANULUS : throw FEMDKNotImplemented(); break;
108  case BGD_BOX : return new Box; break;
109  case BGD_CIRC_HOLE_QUARTER : return new CircHoleQuarter; break;
110  case BGD_CIRCULAR_SECTOR : return new CircularSector; break;
111  case BGD_CSHAPE : return new CShape; break;
112  case BGD_CYLINDER : return new Cylinder; break;
113  case BGD_EXTRUDED : throw FEMDKNotImplemented(); break;
114  case BGD_FIBER_RECTANGLE : return new FiberRectangle; break;
115  case BGD_LSHAPE : return new LShape; break;
116  case BGD_NOTCHED_RECTANGLE : return new NotchedRectangle; break;
117  case BGD_RECTANGLE : return new Rectangle; break;
118  case BGD_SEGMENT_STRIP : return new SegmentStrip; break;
119  //case BGD_SPHERE : return new Sphere; break;
120  case BGD_SPHERE : throw FEMDKNotImplemented(); break;
121 
122  default:
123  FEMDKLOGGER().crit("Internal error: invalid BGD shape type");
124  throw FEMDKException("Internal error: invalid BGD shape type");
125  }
126 }
