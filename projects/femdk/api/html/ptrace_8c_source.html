114 void
115 __NON_INSTRUMENT_FUNCTION__
116 gnu_ptrace(char * what, void * p)

120  FILE * trace ;
121

124  }

129  return ;

132 #ifdef PTRACE_REFERENCE_FUNCTION

136  (void *)GET(DEF,PTRACE_REFERENCE_FUNCTION));
137 #endif
138  first = 0;
139  };
