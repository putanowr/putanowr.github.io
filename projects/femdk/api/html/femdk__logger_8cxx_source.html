104 void setup(const LoggerParams &params) {
105  static bool called = false;
106  if (false == called) {
107  called = true;
108  std::string femdkLoggerName = std::string(FEMDK_LOGGER_NAME);
109  if (params.femdk_parent_logger != "") {
110  if (NULL == log4cpp::Category::exists(params.femdk_parent_logger)) {
111  std::stringstream ss;
112  ss << "The parent logger '" << params.femdk_parent_logger
113  << "' does not exist";
114  FEMDK_ERROR(ss.str());
115  }
116  /* Set parent logger for FEMDK logger */
117  femdkLoggerName = params.femdk_parent_logger + "." + femdkLoggerName;
118  }
119  log_keep = params.log_keep;
120 
121  /* Create logger for FEMDK framework */
122  (void) log4cpp::Category::getInstance(femdkLoggerName);
123  femdkLogger = log4cpp::Category::exists(femdkLoggerName);
124  assert(femdkLogger != NULL);
125 
126  /* Configure the main logger */
127  configureLogger(getLogger(params.femdk_parent_logger), params);
128 
129  getLogger(params.femdk_parent_logger).setPriority(
130  log_level_to_priority(params.log_level));
131 
132  /* Setup priority for FEMDK logger */
133  femdkLogger->setPriority(log_level_to_priority(params.femdk_log_level));
134  }

176 static int create_logfile(const std::string &filename) {
177  static bool called = false;
178  if (false == called) {
179  size_t flen = filename.size();
180  if (flen > logfilenamelen) {
181  std::ostringstream oss;
182  oss << "Log filename to long (> " << logfilenamelen << ")"
183  << "using default: " << logfilename;
184  FEMDK_WARNING(oss.str());
185  } else {
186  strncpy(logfilename, filename.c_str(), logfilenamelen);
187  strcat(logfilename+flen, "XXXXXX");
188  logfilename[flen+6] = '\0';
189  }
190  logfile_id = mkstemp(logfilename);
191  called = true;
192  }
193  return logfile_id;
194 }
