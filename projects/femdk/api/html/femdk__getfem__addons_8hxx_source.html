123  template<typename MAT>
124  void asm_lumped_mass_matrix_global(MAT &M, const mesh_im &mim,
125  const mesh_fem &mf_u1,
126  const mesh_region &rg = mesh_region::all_convexes()) {
127  generic_assembly assem;
128  if (mf_u1.get_qdim() == 1) {
129  assem.set("M(#1,#1)+=sym(comp(Base(#1).Base(#1)));");
130  } else {
131  assem.set("M(#1,#1) +=sym(comp(vBase(#1).vBase(#1))(:,i,:,i));");
132  }
133  assem.push_mi(mim);
134  assem.push_mf(mf_u1);
135  assem.push_mat(M);
136  assem.assembly(rg);
137  for (int i=0; i<gmm::mat_nrows(M); i++) {
138  typename MAT::value_type s = typename MAT::value_type(0);
139  for (int j=0; j<gmm::mat_nrows(M); j++) {
140  if (abs(M(i,j)) >= 1e-10) {
141  s = s+M(i,j);
142  M(i,j) = 0.0;
143  }
144  }
145  M(i,i) = s;
146  }
147  }
