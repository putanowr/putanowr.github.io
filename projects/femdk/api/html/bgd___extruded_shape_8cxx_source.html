41  void ExtrudedShape::export_off(std::ostream &out) const {
42  out << "OFF" << "\n";
43  std::vector<double> coords;
44  myShape->GetBoundaryPoints(coords);
45  int dim = 2;
46  int nvert = coords.size()/dim;
47  int nfaces = 2+nvert;
48  int nedges = 3*nvert; // not used
49  assert(coords.size() == dim*nvert);
50  out << 2*nvert << " " << nfaces << " " << nedges << "\n";
51  std::vector<double>::const_iterator ci = coords.begin();
52  for (int i=0; i<nvert; i++) {
53  out << *ci << " "; ci++;
54  out << *ci << " "; ci++;
55  out << 0.0 << " ";
56  out << "\n";
57  }
58  ci = coords.begin();
59  double dx,dy,dz;
60  GetParam("dx", dx);
61  GetParam("dy", dy);
62  GetParam("dz", dz);
63  for (int i=0; i<nvert; i++) {
64  out << *ci+dx << " "; ci++;
65  out << *ci+dy << " "; ci++;
66  out << dz << " ";
67  out << "\n";
68  }
69  out << nvert << " ";
70  for (int i=0; i<nvert; i++) { out << i << " "; }
71  out << "\n";
72  out << nvert << " ";
73  for (int i=0+nvert; i<nvert+nvert; i++) { out << i << " "; }
74  out << "\n";
75  for (int i=0; i<nvert; i++) {
76  out << 4 << " ";
77  out << i << " " << (i+1)%nvert << " " << (i+1)%nvert+nvert << " " << (i+nvert);
78  out << "\n";
79  }
