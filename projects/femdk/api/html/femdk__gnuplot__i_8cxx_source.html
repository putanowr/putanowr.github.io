764 //------------------------------------------------------------------------------
765 //
766 // Sends a command to an active gnuplot session
767 //
768 Gnuplot& Gnuplot::cmd(const std::string &cmdstr)
769 {
770  if( !(valid) )
771  {
772  return *this;
773  }
774 
775 
776  // int fputs ( const char * str, FILE * stream );
777  // writes the string str to the stream.
778  // The function begins copying from the address specified (str) until it
779  // reaches the terminating null character ('\0'). This final
780  // null-character is not copied to the stream.
781  fputs( (cmdstr+"\n").c_str(), gnucmd );
782 
783  // int fflush ( FILE * stream );
784  // If the given stream was open for writing and the last i/o operation was
785  // an output operation, any unwritten data in the output buffer is written
786  // to the file. If the argument is a null pointer, all open files are
787  // flushed. The stream remains open after this call.
788  fflush(gnucmd);
789 
790 
791  if( cmdstr.find("replot") != std::string::npos )
792  {
793  return *this;
794  }
795  else if( cmdstr.find("splot") != std::string::npos )
796  {
797  two_dim = false;
798  nplots++;
799  }
800  else if( cmdstr.find("plot") != std::string::npos )
801  {
802  two_dim = true;
803  nplots++;
804  }
805 
806  return *this;
807 }

811 //------------------------------------------------------------------------------
812 //
813 // Opens up a gnuplot session, ready to receive commands
814 //
815 void Gnuplot::init()
816 {
817  // char * getenv ( const char * name ); get value of environment variable
818  // Retrieves a C string containing the value of the environment variable
819  // whose name is specified as argument. If the requested variable is not
820  // part of the environment list, the function returns a NULL pointer.
821 #if ( defined(unix) || defined(__unix) || defined(__unix__) ) && !defined(__APPLE__)
822  if (getenv("DISPLAY") == NULL)
823  {
824  valid = false;
825  throw GnuplotException("Can't find DISPLAY variable");
826  }
827 #endif
828 
829 
830  // if gnuplot not available
831  if (!Gnuplot::get_program_path())
832  {
833  valid = false;
834  throw GnuplotException("Can't find gnuplot");
835  }
836 
837 
838  //
839  // open pipe
840  //
841  std::string tmp = Gnuplot::m_sGNUPlotPath + "/" +
842  Gnuplot::m_sGNUPlotFileName;
843 
844  // FILE *popen(const char *command, const char *mode);
845  // The popen() function shall execute the command specified by the string
846  // command, create a pipe between the calling program and the executed
