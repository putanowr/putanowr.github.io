52 GeomPtr
53 GeomFactory::getGeom(const std::string &geomTypeName) const {
54  if (false == checkIfSupported(geomTypeName)) {
55  return GeomPtr();
56  }
57  geom_type_t mt = str2geom_type(geomTypeName);
58  switch (mt) {
59  case GEOM_BGD: {
60  std::str shapeName = geomTypeName.substr(4);
61  return GeomPtr(new GeomBGD(shapeName));
62  break;
63  }
64  default:
65  FEMDKLOGGER().crit("Internal error: invalid geom model type");
66  throw FEMDKException("Internal error: invalid geom model type");
67  }
68 }
