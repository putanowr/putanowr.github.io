77 void merge_meshes(getfem::mesh &dest, const getfem::mesh &source) {
78  /* check if source is the same as the destination */
79  if (&dest == &source) {
80  return;
81  }
82 
83  getfem::size_type source_nb_points = source.nb_points();
84 
85  /* vector for mapping between point indices in the source mesh
86  * and point indices in the destination mesh
87  */
88  std::vector<getfem::size_type> dest_index(source_nb_points);
89 
90  /* add all points from the source mesh to the destination mesh building
91  * a mapping between point indices
92  */
93  getfem::mesh::PT_TAB const &source_points = source.points();
94  for(dal::bv_visitor s(source.points_index()); !s.finished(); ++s) {
95  dest_index[s] = dest.add_point(source_points[s]);
96  }
97  /* add all convexes from the source mesh to the destination mesh
98  */
99  /* iterate on source convexes and and them to dest mesh building on the fly
100  * convex to nodes incidence relations
101  */
102  for(dal::bv_visitor cv(source.convex_index()); !cv.finished(); ++cv) {
103  bgeot::pgeometric_trans source_geom_trans =
104  source.trans_of_convex(cv);
105  getfem::mesh::ind_cv_ct source_cv_points = source.ind_points_of_convex(cv);
106  int cv_nb_points = source_cv_points.size();
107  std::vector<getfem::size_type> dest_cv_points(cv_nb_points);
108  /* map the source point indices to destination point indices for the convex
109  * cv
110  */
111  for (int i=0; i<cv_nb_points; ++i) {
112  dest_cv_points[i] = dest_index[source_cv_points[i]];
113  }
114  dest.add_convex(source_geom_trans, dest_cv_points.begin());
115  }
116 } /* end of merge_meshes() */

153 void export_mesh_raw_coords(const getfem::mesh &mesh, std::ostream &ost) {
154  dal::bit_vector nn = mesh.convex_index();
155  bgeot::size_type i;
156  for (i << nn; i != bgeot::size_type(-1); i << nn) {
157  getfem::mesh::ref_mesh_pt_ct pts = mesh.points_of_convex(i);
158  if (pts.size() == 4) {
159  ost << pts[0][0] << " " << pts[0][1] << "\n";
160  ost << pts[1][0] << " " << pts[1][1] << "\n";
161  ost << pts[3][0] << " " << pts[3][1] << "\n";
162  ost << pts[2][0] << " " << pts[2][1] << "\n";
163  ost << pts[0][0] << " " << pts[0][1] << "\n";
164  ost << "\n";
165  } else {
166  for (bgeot::size_type j=0; j<pts.size(); j++) {
167  ost << pts[j][0] << " " << pts[j][1] << "\n";
168  }
169  ost << pts[0][0] << " " << pts[0][1] << "\n";
170  ost << "\n";
171  }
172  }

217 void write_points(const getfem::mesh &mesh,
218  std::ostream &ost, PointBasicFormater &fmt) {
219  ost.precision(16);

230 void write_convexes(const getfem::mesh &mesh,
231  std::ostream &ost, ConvexBasicFormater &fmt) {
232  ost.precision(16);
233  gmm::stream_standard_locale sl(ost);
234  fmt.write_header(ost);
235  for (dal::bv_visitor i(mesh.convex_index()); !i.finished(); ++i) {
236  bgeot::pgeometric_trans gt = mesh.trans_of_convex(i);
237  std::string description = bgeot::name_of_geometric_trans(gt);
238  int dim = gt->dim();
239  int nvert = gt->nb_points();
240 
241  /* verts is a container holding indices of convex vertices */
242  const getfem::mesh::ind_cv_ct& verts = mesh.ind_points_of_convex(i);
243  fmt.write_convex(ost, i, verts.begin(), verts.end(),
244  dim, nvert, description, true);
245  }
246  fmt.write_footer(ost);
247 }
