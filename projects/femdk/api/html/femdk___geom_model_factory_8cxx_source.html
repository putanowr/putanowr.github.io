49 GeomModelFactory::getModel(const ConfigData &geomConfig,
50  std::string& modelTypeName) const {
51  modelTypeName = getString(geomConfig, "model-type", "model-not-specified");
52 
53  if (false == checkIfSupported(modelTypeName)) {
54  return NULL;
55  }
56 
57  geom_type_t mt = str2geom_type(modelTypeName);
58 
59  switch (mt) {
60  case GEOM_BGD:
61  return new GeomModelBGD;
62  break;
63  case GEOM_GMSH:
64 #ifdef FEMDK_HAVE_GMSH
65  return new GeomModelGmsh;
66 #else
67  FEMDKLOGGER().crit("Internal error: not configured with gmsh");
68  throw FEMDKException("Internal error: not configured with gmsh");
69 #endif
70  break;
71  case GEOM_IGEOM:
72 #ifdef FEMDK_HAVE_MOAB
73  return new GeomModeliGeom;
74 #else
75  FEMDKLOGGER().crit("Build error: not configured with MOAB");
76  throw FEMDKException("Build error: not configured with MOAB");
77 #endif
78  break;
79  default:
80  FEMDKLOGGER().crit("Internal error: invalid geom model type");
81  throw FEMDKException("Internal error: invalid geom model type");
82  }
83 }
