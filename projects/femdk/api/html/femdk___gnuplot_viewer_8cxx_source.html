57 void GnuplotViewer::setAxis(const std::vector<double> &coords) {
58  int s = coords.size();
59  if (2 == s) {
60  set_xrange(coords[0], coords[1]);

62  set_xrange(coords[0], coords[2]);

69  throw FEMDKException("Invalid number of coordinates for bounding box");

144  s << "set rmargin screen " << 1-0.05 - myHmargin << "\n";
145  s << "set bmargin screen " << 0.05 << "\n";
146  s << "set tmargin screen " << 0.05 + myVmargin << "\n";
147  s << "set yrange [ * : * ] reverse " << "\n";
148  s << "plot '-' notitle \n";
149  s << xp[0] << " " << data[0] << "\n";
150  s << xp[1] << " " << data[1] << "\n";
151  s << "e\n";

231 GnuplotViewer::fitLinear(const bgd::Rectangle &rectangle, int side,
232  int size, double *data, bool plot_on_side) {
233  switch (side) {
234  case 1: /* south */
235  plot_on_south(rectangle, size, data);
236  break;
237  case 2: /* east */
238  plot_on_east(rectangle, size, data);
239  break;
240  case 3: /* north */
241  plot_on_north(rectangle, size, data);
242  break;
243  case 4: /* west */
244  plot_on_west(rectangle, size, data);
245  break;
246  default :
247  std::ostringstream ss;
248  ss << "Invalid side number : " << side;
249  throw FEMDKInvalidArgs(ss.str());
250  }
251 }
