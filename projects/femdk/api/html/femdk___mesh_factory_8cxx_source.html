43 MeshFactory::getMesh(const std::string &meshTypeName) const {
44  if (false == checkIfSupported(meshTypeName)) {
45  return MeshPtr();
46  }
47  mesh_type_t mt = str2mesh_type(meshTypeName);
48  switch (mt) {
49  case MESH_GETFEM:
50  return MeshPtr(new MeshGetfem);
51  break;
52  default:
53  FEMDKLOGGER().crit("Internal error: invalid mesh model type");
54  throw FEMDKException("Internal error: invalid mesh model type");
55  }
56 }
