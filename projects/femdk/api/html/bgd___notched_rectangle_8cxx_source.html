88  void NotchedRectangle::export_geo(std::ostream &out) const {
89  double b = GetLigamentLength();
90  double Ox,Oy, Lx, Ly;
91  GetOrigin(Ox,Oy);
92  GetParam("Lx", Lx);
93  GetParam("Ly", Ly);
94  int Nb, Nn;
95  GetParam("Nb", Nb);
96  GetParam("Nn", Nn);
97  std::vector<double> lc;
98  GetParam("lc", lc);
99  if (lc.size() != Nb+Nn+3) {
100  std::cerr << "WARNING : \n";
101  double lcambient, lcnotch;
102  GetParam("lcambient", lcambient);
103  GetParam("lcnotch", lcnotch);
104  lc.resize(Nb+Nn+3, lcambient);
105  lc[Nb-1] = (lcnotch+lcambient)/2.0;
106  lc[Nb]=lcnotch;
107  lc[Nb+1] = (lcnotch+lcambient)/2.0;
108  }
109  out << "Ox = " << Ox << ";\n";
110  out << "Oy = " << Oy << ";\n";
111  out << "Lx = " << Lx << ";\n";
112  out << "Ly = " << Ly << ";\n";
113  out << "b = " << b << ";\n";
114  out << "Nb = " << Nb << ";\n";
115  out << "Nn = " << Nn << ";\n";
116  out << "dxb = b/Nb;\n";
117  out << "dxn = (Lx-b)/Nn;\n";
118  out << "__stp = newp;\n";
119  int j=0;
120  out << "Point(__stp) = {Ox, Oy, 0," << lc[j++] << "}; // O \n";
121 
122  for (int i=1; i<Nb+1; i++) {
123  out << "Point(newp) = {Ox+" << i << "*dxb, Oy, 0," << lc[j++] << "}; // ligament\n";
124  }
125  for (int i=1; i<Nn; i++) {
126  out << "Point(newp) = {Ox+b+" << i << "*dxn, Oy, 0," << lc[j++] << "}; // notch\n";
127  }
128  out << "Point(newp) = {Ox+Lx,Oy, 0," << lc[j++] << "}; // south-east corner \n";
129  out << "Point(newp) = {Ox+Lx,Oy+Ly,0," << lc[j++] << "}; // north-east corner\n";
130  out << "Point(newp) = {Ox, Oy+Ly,0," << lc[j++] << "}; // nort-west corner \n";
131  out << "\n";
132  out << "__stl= newl;\n";
133  out << "__ln = __stl;\n";
134  out << "outl[] = {};\n";
135  out << "ligl[] = {};\n";
136  out << "notchl[] = {};\n";
137  out << "For i In {0: Nb-1}\n";
138  out << "Line(__ln) = {__stp+i, __stp+i+1};\n";
139  out << "outl[] += __ln;\n";
140  out << "ligl[] += __ln;\n";
141  out << "__ln = newl;\n";
142  out << "EndFor\n";
143  out << "For i In {Nb: Nb+Nn-1}\n";
144  out << "Line(__ln) = {__stp+i, __stp+i+1};\n";
145  out << "outl[] += __ln;\n";
146  out << "notchl[] += __ln;\n";
147  out << "__ln = newl;\n";
148  out << "EndFor\n";
149  out << "For i In {Nb+Nn: Nb+Nn+3-1}\n";
150  out << "Line(__ln) = {__stp+i, __stp+(i+1)%(Nb+Nn+3)};\n";
151  out << "outl[] += __ln;\n";
152  out << "__ln = newl;\n";
153  out << "EndFor\n";
154  out << "__perim = newll;\n";
155  out << "Line Loop(__perim) = {outl[]};\n";
156  out << "\n";
157  out << "__sur = news;\n";
158  out << "Plane Surface(__sur) = {__perim};\n";
159  out << "\n";
160  out << "Physical Line(\"ligament\")= {ligl[]};\n";
161  out << "Physical Line(\"notch\") = {notchl[]};\n";
162  out << "Physical Line(\"east\")= {__stl+Nb+Nn+1};\n";
163  out << "Physical Line(\"north\")= {__stl+Nb+Nn+2};\n";
164  out << "Physical Line(\"west\")= {__stl+Nb+Nn+3};\n";
165  out << "Physical Surface(\"all\") = {__sur};\n";
166  }

243  BGDErrorCode
244  NotchedRectangle::GetBoundaryPoints(std::vector<double> &coords, bool
245  repeat_first) const {
246  double b = GetLigamentLength();
247  double Ox,Oy, Lx, Ly;
248  int Nx, Ny, Nn, Nb;
249  GetOrigin(Ox,Oy);
250  GetParam("Lx", Lx);
251  GetParam("Ly", Ly);
252  GetParam("Nx", Nx);
253  GetParam("Ny", Ny);
254  GetParam("Nb", Nb);
255  GetParam("Nn", Nn);
256 
257  assert(1 <= Nx);
258  assert(1 <= Ny);
259  assert(1 <= Nn);
260  assert(1 <= Nb);
261  double dx = Lx/Nx;
262  double dy = Ly/Ny;
263 
264  /* bottom edge */
265  double dxb, dxn;
266  dxb = b/Nb;
267  dxn = (Lx-b)/Nn;
268 
269  double x = Ox;
270  double y = Oy;
271 
272  coords.push_back(x);
273  coords.push_back(y);
274 
275  for (int i=0; i<Nb; i++) {
276  x = x + dxb;
277  coords.push_back(x);
278  coords.push_back(y);
279  }
280  for (int i=1; i<Nn; i++) {
281  x = x + dxn;
282  coords.push_back(x);
283  coords.push_back(y);
284  }
285 
286  x = Ox+Lx;
287  coords.push_back(x);
288  coords.push_back(y);
289  /* right edge */
290  for (int i=1; i<Ny+1; i++) {
291  y = Oy + i*dy;
292  coords.push_back(x);
293  coords.push_back(y);
294  }
295  /* top edge */
296  for (int i=Nx-1; i>=0; i--) {
297  x = Ox + i *dx;
298  coords.push_back(x);
299  coords.push_back(y);
300  }
301 
302  for (int i=Ny-1; i>0; i--) {
303  y = Oy + i *dy;
304  coords.push_back(x);
305  coords.push_back(y);
306  }
307 
308  if (true == repeat_first) {
309  x = Ox;
310  y = Oy;
311  coords.push_back(x);
312  coords.push_back(y);
313  }
314 
315  return BGD_SUCCESS;
316  }
