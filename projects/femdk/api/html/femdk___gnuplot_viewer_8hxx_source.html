97 void GnuplotViewer::showFemField(const getfem::mesh_fem &fem,
98  const T &dofs) {
99  getfem::size_type N = fem.nb_dof();
100  assert(dofs.size() == N);
101  T x(N);
102  T y(N);
103  for (getfem::size_type i=0; i<N; i++) {
104  bgeot::base_node p = fem.point_of_basic_dof(i);
105  x[i] = p[0];
106  y[i] = p[1];
107  }
108  plot_xyz(x,y,dofs);
109 }
