42  void SegmentStrip::export_geo(std::ostream &out) const {
43  double Ly,Ox,Oy;
44  int Nx,Ny;
45  std::vector<double> lc;
46  std::vector<double> dLxVec;
47  GetParam("Ly", Ly);
48  GetParam("Nx", Nx);
49  GetParam("Ny", Ny);
50  GetParam("Ox", Ox);
51  GetParam("Oy", Oy);
52  GetParam("lc", lc);
53  GetParam("dLxVec", dLxVec);
54  bool internal_lines = HasInternalLines();
55  out << "Ly = " << Ly << ";\n";
56  out << "Ox = " << Ox << ";\n";
57  out << "Oy = " << Oy << ";\n";
58  out << "nsegments = " << dLxVec.size() << ";\n";
59  out << "internal_lines = " << internal_lines << ";\n";
60  out << "stp = newp;\n";
61  out << "lc[] = {";
62  int i;
63  for (i=0; i<lc.size()-1; i++) {
64  out << lc[i] << ",";
65  }
66  out << lc[i] << "};\n";
67  double Lxi = 0;
68  out << "Lxi[] = {";
69  for (i=0; i<dLxVec.size(); i++) {
70  out << Lxi << ",";
71  Lxi += dLxVec[i];
72  }
73  out << Lxi << "};\n";
74  out << "For i In {1:nsegments+1}\n"
75  " Point(newp) = {Ox + Lxi[i-1], Oy, 0, lc[i-1]}; \n"
76  "EndFor\n";
77  out << "For i In {1:nsegments+1}\n"
78  " Point(newp) = {Ox + Lxi[nsegments+1-i], Oy+Ly, 0, lc[i-1]}; \n"
79  "EndFor\n";
80  out << "outl[] = {};\n";
81  out << "stl = newl;\n";
82  /* South lines */
83  out << "For i In {0: nsegments-1}\n"
84  " __ln = newl;\n"
85  " Line(__ln) = {stp+i,stp + (i+1)};\n"
86  " outl[] += __ln;\n"
87  " Physical Line(Sprintf(\"south%g\",i+1)) = {__ln};\n"
88  "EndFor\n";
89  out << " __ln = newl;\n"
90  " Line(__ln) = {stp+i,stp + (i+1)};\n"
91  " outl[] += __ln;\n"
92  " Physical Line(\"east\") = {__ln};\n";
93  out << "For i In {0: nsegments-1}\n"
94  " __ln = newl;\n"
95  " Line(__ln) = {stp+i+nsegments+1,stp + (i+nsegments+2)};\n"
96  " outl[] += __ln;\n"
97  " Physical Line(Sprintf(\"north%g\",nsegments-i)) = {__ln};\n"
98  "EndFor\n";
99  out << " __ln = newl;\n"
100  " Line(__ln) = {stp+2*nsegments+1,stp};\n"
101  " outl[] += __ln;\n"
102  " Physical Line(\"west\") = {__ln};\n";
103  out << "If (0 != internal_lines)\n"
104  "For i In {1: nsegments-1}\n"
105  " __ln = newl;\n"
106  " Line(__ln) = {stp+2*nsegments+1-i, stp+i};\n"
107  " Physical Line(Sprintf(\"interline%g\",i)) = {__ln};\n"
108  "EndFor\n"
109  "EndIf\n";
110  out << "If (0 == internal_lines)\n"
111  " perim = newll;\n"
112  " Line Loop(perim) = {outl[]};\n"
113  " sur = news;\n"
114  " Plane Surface(sur) = {perim};\n"
115  " Physical Surface(\"segment1\") = {sur};\n";
116  out << "EndIf\n"
117  "If (0 != internal_lines)\n"
118  "ns = nsegments;\n"
119  "For i In {0:nsegments-2}\n"
120  " __ll = newll;\n"
121  " Line Loop(__ll) = { stl+i, -(stl+2*ns+2+i), stl+2*ns-i, stl+2*ns+1+i };\n"
122  " sur = news;\n"
123  " Plane Surface(sur) = {__ll};\n"
124  " Physical Surface(Sprintf(\"segment%g\",i+1)) = { sur };\n"
125  "EndFor\n"
126  "__ll = newll;\n"
127  " Line Loop(__ll) = {stl+ns-1, stl+ns, stl+ns+1, stl+2*ns+ns};\n"
128  " sur = news;\n"
129  " Plane Surface(sur) = {__ll};\n"
130  " Physical Surface(Sprintf(\"segment%g\",i+1)) = { sur };\n"
131  "EndIf\n";
132  return;
133  }

135  void SegmentStrip::export_oogl(std::ostream &out) const {
136  double Ly,Ox,Oy;
137  std::vector<double> dLxVec;
138  GetParam("Ly", Ly);
139  GetParam("Ox", Ox);
140  GetParam("Oy", Oy);
141  GetParam("dLxVec", dLxVec);
142  bool internal_lines = HasInternalLines();
143  out << "OFF" << "\n";
144  std::vector<double> coords;
145  GetBoundaryPoints(coords);
146  int nvert = coords.size()/myDim;
147  int nfaces = internal_lines ? GetNumOfSegments() : 1;
148  int nedges = nvert; // not used
149  assert(coords.size() == myDim*nvert);
150  out << nvert << " " << nfaces << " " << nedges << "\n";
151  std::vector<double>::const_iterator ci = coords.begin();
152  for (int i=0; i<nvert; i++) {
153  out << *ci << " "; ci++;
154  out << *ci << " "; ci++;
155  out << 0.0 << " ";
156  out << "\n";
157  }
158  if (internal_lines) {
159  for (int i = 0; i<nfaces; i++) {
160  out << 4 << " ";
161  out << i << " " << i+1 << " "<< nvert-2-i << " " << nvert-1-i;
162  out << "\n";
163  }
164  } else {
165  out << nvert << " ";
166  for (int i=0; i<nvert; i++) {
167  out << i << " ";
168  }
169  }
170  out << "\n";
171  }

194  void SegmentStrip::SetLCFromSegments(double scale) {
195  double lc;
196  int n = GetNumOfVertices();
197  std::vector<double> nlc(n, lc);
198  std::vector<double> dLxVec;
199  GetParam("dLxVec", dLxVec);
200  nlc[0] = nlc[n-1] = scale*dLxVec[0];
201  for (int i=0; i<dLxVec.size()-1; i++) {
202  double aver = sqrt(dLxVec[i] * dLxVec[i+1]);
203  //double aver = (dLxVec[i] + dLxVec[i+1])/2.0;
204  lc = scale * aver;
205  nlc[i+1] = lc;
206  nlc[n-2-i] = lc;
207  }
208  lc = scale*dLxVec[dLxVec.size()-1];
209  nlc[n/2-1] = nlc[n/2] = lc;
210  SetParam("lc", nlc);
211  }

255  BGDErrorCode
256  SegmentStrip::GetBoundaryPoints(std::vector<double> &coords, bool
257  repeat_first) const {
258  double Ly,Ox,Oy;
259  std::vector<double> dLxVec;
260  GetParam("Ly", Ly);
261  GetParam("Ox", Ox);
262  GetParam("Oy", Oy);
263  GetParam("dLxVec", dLxVec);
264  coords.push_back(Ox);
265  coords.push_back(Oy);
266  double x=Ox;
267  for (int i=0;i<dLxVec.size(); i++) {
268  x+=dLxVec[i];
269  coords.push_back(x);
270  coords.push_back(Oy);
271  }
272  coords.push_back(x);
273  coords.push_back(Oy+Ly);
274  for (int i=dLxVec.size()-1; i>0; i--) {
275  x-=dLxVec[i];
276  coords.push_back(x);
277  coords.push_back(Oy+Ly);
278  }
279  coords.push_back(Ox);
280  coords.push_back(Oy+Ly);
281  if (repeat_first) {
282  coords.push_back(Ox);
283  coords.push_back(Oy);
284  }
285  }

324  void SegmentStrip::GetVertex(int i, double &xout, double &yout) const {
325  double Lx, Ly;
326  Lx = GetLx();
327  GetParam("Ly", Ly);
328  double x,y;
329  GetParam("Ox",x);
330  GetParam("Oy",y);
331  switch (i%4) {
332  case 0:
333  xout = x;
334  yout = y;
335  break;
336  case 1:
337  xout = x+Lx;
338  yout = y+0.0;
339  break;
340  case 2:
341  xout = x+Lx;
342  yout = y+Ly;
343  break;
344  case 3:
345  xout = x+0.0;
346  yout = y+Ly;
347  break;
348  default:
349  throw FEMDKException("Internal error -- impossible condition");
350  }
