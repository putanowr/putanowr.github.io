67  template<class ITER>
68  void write_record(std::ostream &ost, ITER b, ITER e) {
69  ost << format_prefix();
70  if (b != e) {
71  ost << *b;
72  b++;
73  }
74  std::string separator = format_separator();
75  for ( ; b != e; ++b) {
76  ost << separator << *b;
77  }
78  ost << format_suffix();
79  }

159  template<class ITER>
160  void write_point(std::ostream &ost, int i,
161  ITER b, ITER e, bool valid=true) {
162  if (false != valid && false != do_not_validate) {
163  return;
164  }
165  cur_index = i;
166  int n = myDim < 3 ? 3 : myDim;
167  if (true == fill_coords && std::distance(b,e) < n) {
168  std::vector<double> v(n, 0.0);
169  std::copy(b, e, v.begin());
170  write_record(ost, v.begin(), v.end());
171  } else {
172  write_record(ost, b, e);
173  }
174  }

272 ConvexTochnogFormater::format_prefix() {
273  if (false == validate_convex_data(cur_descrip, cur_nvert)) {
274  std::ostringstream oss;
275  oss << "Invalid datasize for tochnog element";
276  oss << " '" << cur_descrip << "'";

278  throw FEMDKException(oss.str());
279  }
280  std::ostringstream oss;

282  oss << " " << cur_index << " ";
283  if (NULL != descripMapper) {
