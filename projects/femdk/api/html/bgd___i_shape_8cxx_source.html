48  void IShape::export_geo(std::ostream &out) const {
49  double t,b,Sb,St,vx,vy,g,H,Ox,Oy;
50  std::vector<double> lc(GetNumOfVertices(),1.0);
51  GetParam("H", H);
52  GetParam("Sb", Sb);
53  GetParam("St", St);
54  GetParam("b", b);
55  GetParam("t", t);
56  GetParam("g", g);
57  GetParam("vx", vx);
58  GetParam("vy", vy);
59  GetParam("Ox", Ox);
60  GetParam("Oy", Oy);
61  GetParam("lc", lc);
62  out << "H = " << H << ";\n";
63  out << "Sb = " << Sb << ";\n";
64  out << "St = " << St << ";\n";
65  out << "g = " << g << ";\n";
66  out << "b = " << b << ";\n";
67  out << "t = " << t << ";\n";
68  out << "vx = " << vx << ";\n";
69  out << "vy = " << vy << ";\n";
70  out << "Ox = " << Ox << ";\n";
71  out << "Oy = " << Oy << ";\n";
72  out << "lc[] = {";
73  {
74  int i=0;
75  for (; i<lc.size()-1; i++) {
76  out << lc[i] << ",";
77  }
78  out << lc[i] << "};\n";
79  }
80 
81  out << "stp = newp;\n"
82  "Point(stp) = {Ox, Oy, 0, lc[0]};\n"
83  "Point(newp) = {Ox+0.5*Sb, Oy, 0, lc[1]};\n"
84  "Point(newp) = {Ox+0.5*Sb, Oy+b, 0, lc[2]};\n"
85  "Point(newp) = {Ox+0.5*g+vx, Oy+b, 0, lc[3]};\n"
86  "Point(newp) = {Ox+0.5*g, Oy+b+vy, 0, lc[4]};\n"
87  "Point(newp) = {Ox+0.5*g, Oy+H-t-vy, 0, lc[5]};\n"
88  "Point(newp) = {Ox+0.5*g+vx, Oy+H-t, 0, lc[6]};\n"
89  "Point(newp) = {Ox+0.5*St, Oy+H-t, 0, lc[7]};\n"
90  "Point(newp) = {Ox+0.5*St, Oy+H, 0, lc[8]};\n"
91  "Point(newp) = {Ox, Oy+H, 0, lc[9]};\n"
92  /* half of the space */
93  "Point(newp) = {Ox-0.5*St, Oy+H, 0, lc[10]};\n"
94  "Point(newp) = {Ox-0.5*St, Oy+H-t, 0, lc[11]};\n"
95  "Point(newp) = {Ox-0.5*g-vx, Oy+H-t, 0, lc[12]};\n"
96  "Point(newp) = {Ox-0.5*g, Oy+H-t-vy, 0, lc[13]};\n"
97  "Point(newp) = {Ox-0.5*g, Oy+b+vy, 0, lc[14]};\n"
98  "Point(newp) = {Ox-0.5*g-vx, Oy+b, 0, lc[15]};\n"
99  "Point(newp) = {Ox-0.5*Sb, Oy+b, 0, lc[16]};\n"
100  "Point(newp) = {Ox-0.5*Sb, Oy, 0, lc[17]};\n";
101  out << "\n";
102  out << "outl[] = {};\n";
103  out << "For i In {0:17}\n"
104  " __ln = newl;\n"
105  " Line(__ln) = {stp+i, stp+Fmod(i+1,18)};\n"
106  " outl[] += __ln;\n"
107  "EndFor\n"
108  "\n";
109  out << "perim = newll;\n"
110  "Line Loop(perim) = {outl[]};\n"
111  "\n"
112  " sur = news;\n"
113  "Plane Surface(sur) = {perim};\n"
114  "\n";
115  for (int i=0; i<18;i++) {
116  out << "Physical Line(\"b" << i+1 << "\")= {outl[" << i << "]};\n";
117  }
118  out << "Physical Surface(\"all\") = {sur};\n";
119  }
