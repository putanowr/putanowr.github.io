110 GridPtr
111 VTKExporter::transferMeshStructure(const getfem::mesh &mesh) {
112  GridPtr gp = GridPtr::New();
113 
114  vtkPoints *points = vtkPoints::New();
115  points->SetNumberOfPoints(mesh.nb_points());
116  std::vector<int> dofmap(pmf->nb_dof());
117  int cnt = 0;
118  vtkIdType id=0;
119  for (dal::bv_visitor d(pmf_dof_used); !d.finished(); ++d) {
120  dofmap[d] = cnt++;
121  getfem::base_node P = pmf->point_of_basic_dof(d);
122 
123  double xyz[3];
124  set_coords(P, xyz);
125  points->SetPoint(id, xyz);
126  id++;
127  }
128  gp->SetPoints(points);
129  vtkIdType ptIds[32];
130  for (dal::bv_visitor cv(pmf->convex_index()); !cv.finished(); ++cv) {
131 #if defined(GETFEM_5_2)
132  const std::vector<unsigned> &dmap = getfem_to_vtk_dof_mapping(pmf_mapping_type[cv]);
133  int type = pmf_mapping_type[cv];
134 #else
135  const std::vector<unsigned> &dmap = getfem_to_vtk_dof_mapping(pmf_cell_type[cv]);
136  int type = pmf_cell_type[cv];
137 #endif
138  vtkIdType npts = dmap.size();
139  for (getfem::size_type i=0; i < dmap.size(); ++i) {
140  ptIds[i] = (vtkIdType(dofmap[pmf->ind_basic_dof_of_element(cv)[dmap[i]]]));
141  }
142  gp->InsertNextCell(type, npts, ptIds);
143  }
144  return gp;
145 }
