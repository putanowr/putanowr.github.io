131  logging::getFEMDKLogger().debug("Creating VTK visualisation pipe");
132  // The usual renderer, render window and interactor
133  myRenderer = vtkSmartPointer<vtkRenderer>::New();
134  myRenderWindow = vtkSmartPointer<vtkRenderWindow>::New();
135  myInteractor = vtkSmartPointer<vtkRenderWindowInteractor>::New();
136 
137  myRenderer->SetBackground(.1, .2, .3);
138  myRenderWindow->AddRenderer(myRenderer);
139  myInteractor->SetRenderWindow(myRenderWindow);
140  logging::getFEMDKLogger().debug("Sucessfuly created pipe");
141  }

148 void VTKViewer::buildVTKPipeForMesh(const std::string &name) {
149  logging::getFEMDKLogger().debug("Building pipe for mesh");
150  MapperPtr mapper = MapperPtr::New();
151  ActorPtr actor = ActorPtr::New();
152  GridItr gi = myGrids.find(name);
153  if(gi != myGrids.end()) {
154  mapper->SetInputData(gi->second);
155  actor->SetMapper(mapper);
156  myRenderer->AddActor(actor);
157  myActors[name] = actor;
158  logging::getFEMDKLogger().debug("Creating VTK visualisation pipe");
159  } else {
160  std::ostringstream oss;
161  oss << "Cannot find grid named: " << name;
162  throw FEMDKException(oss.str());
163  }
164 }

205 bool VTKViewer::showCellScalars(const std::string &meshname,
206  const std::string &arrayname,
207  double *range) {
208  bool result = false;
209  GridItr gi = myGrids.find(meshname);
210  if(gi != myGrids.end()) {
211  UGridPtr gp = gi->second;
212  vtkCellData *cd = gp->GetCellData();
213  if (cd->HasArray(arrayname.c_str())) {
214  cd->SetActiveScalars(arrayname.c_str());
215  vtkDataSetMapper *mapper = static_cast<vtkDataSetMapper*>(myActors[meshname]->GetMapper());
216  mapper->SetScalarModeToUseCellData();
217  mapper->SetScalarVisibility(1);
218  if (NULL != range) {
219  mapper->SetScalarRange(range[0], range[1]);
220  } else {
221  double data[2];
222  cd->GetArray(arrayname.c_str())->GetRange(data);
223  mapper->SetScalarRange(data);
224  }
225  myRenderWindow->Render();
226  result = true;
227  }
228  }
229  return result;
230 }

275 BitArrayPtr
276 VTKViewer::addBitCellArray(const std::string &meshname,
277  const std::string &arrayname,
278  bool defaultValue) {
279  GridItr gi = myGrids.find(meshname);
280  if(gi != myGrids.end()) {
281  UGridPtr gp = gi->second;
282  vtkCellData *cd = gp->GetCellData();
283  if (cd->HasArray(arrayname.c_str())) {
284  return NULL;
285  } else {
286  vtkBitArray *array = vtkBitArray::New();
287  array->SetName(arrayname.c_str());
288  array->SetNumberOfComponents(1);
289  vtkIdType nc = gp->GetNumberOfCells();
290  array->SetNumberOfValues(nc);
291  for (vtkIdType i=0; i<nc; i++) {
292  array->SetValue(i, defaultValue);
293  }
294  cd->AddArray(array);
295  return array;
296  }
297  } else {
298  std::ostringstream oss;
299  oss << "Could not find grid named: " << meshname;
300  logging::getFEMDKLogger().info(oss.str());
301  return NULL;
302  }
303 }
