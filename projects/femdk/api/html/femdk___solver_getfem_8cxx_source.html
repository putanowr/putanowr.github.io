171 SolverGetfem::createFEM(const std::string& name,

173  std::string *true_fem_name) {
174  MeshPtr mesh = getMesh(mesh_name);
175  MeshFemPtr fem(new ::getfem::mesh_fem(*mesh));
176  std::string refname = nextFEMName(name);
177  addFEMName(name);
