248 void make_test_mesh(const std::string& name, getfem::mesh &mesh) {
249  mesh.clear();
250  TestMeshInfo tmi;
251  tmi = get_test_mesh_info(name);
252  const double *pt = tmi.coords;
253  for (int i=0; i<tmi.info[1]; ++i) {
254  if (2 == tmi.info[0]) {
255  mesh.add_point(bgeot::base_node(*pt, *(pt+1)));
256  } else if ( 3 == tmi.info[0]) {
257  mesh.add_point(bgeot::base_node(*pt, *(pt+1), *(pt+2)));
258  } else {
259  std::stringstream oss;
260  oss << "Invalid geometric dimension '" << tmi.info[0] <<"'"
261  << " for test mesh '" << name << "'";
262  FEMDK_ERROR(oss.str().c_str());
263  }
264  pt += tmi.info[0];
265  }
266  const int *ct = tmi.cells;
267  for (int i=0; i<tmi.info[2]; ++i) {
268  bgeot::pgeometric_trans pgt;
269  pgt = bgeot::geometric_trans_descriptor(tmi.geotrans[tmi.celltypes[i]]);
270  mesh.add_convex(pgt, ct+1); // ct+1 is the pointer to the firs node index
271  ct += ct[0]+1; // ct[0] holds the number of nodes pre cell
272  }
273 }
