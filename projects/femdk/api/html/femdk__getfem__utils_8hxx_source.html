52 template <class VECTOR>
53 direction_2D_t classify_normal_2D(const VECTOR &un,
54  const double tolerance=1e-8) {
55  double x = un[0];
56  double y = un[1];
57 
58  GMM_ASSERT1(x*x+y*y >= tolerance, "Zero (or nearly) vector");
59 
60  if (fabs(x) < tolerance) {
61  if (y > 0) {
62  return DIR_N;
63  } else {
64  return DIR_S;
65  }
66  }
67  if (fabs(y) < tolerance) {
68  if (x > 0) {
69  return DIR_E;
70  } else {
71  return DIR_W;
72  }
73  }
74  if (x > 0.0 && y > 0.0) {
75  return DIR_NE;
76  } else if (x < 0.0 && y > 0.0) {
77  return DIR_NW;
78  } else if (x < 0.0 && y < 0.0) {
79  return DIR_SW;
80  } else if (x > 0.0 && y < 0.0) {
81  return DIR_SE;
82  }
83  return DIR_UNDEFINED;
84 }
