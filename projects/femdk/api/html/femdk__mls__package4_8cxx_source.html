198  dal::bit_vector decimate(const mesher_signed_distance &ref_element,
199  scalar_type dmin) const {
200  dal::bit_vector retained_points;
201  if (size() != 0) {
202  size_type n = points[0].size();
203 
204  bgeot::kdtree points_tree;
205  points_tree.reserve(size());
206  for (size_type i = 0; i < size(); ++i)
207  points_tree.add_point_with_id(points[i], i);
208 
209  for (size_type nb_co = n; nb_co != size_type(-1); nb_co --) {
210  for (size_type i = 0; i < size(); ++i) {
211  if (!(retained_points.is_in(i)) &&
212  (constraints(i).card() == nb_co ||
213  (nb_co == n && constraints(i).card() > n)) &&
214  ref_element(points[i]) < 1E-8) {
215  bool kept = true;
216  scalar_type d = (nb_co == 0) ? (dmin * 1.5)
217  : std::min(radius(i)*0.25, dmin);
218  base_node min = points[i], max = min;
219  for (size_type m = 0; m < n; ++m) { min[m]-=d; max[m]+=d; }
220  bgeot::kdtree_tab_type inpts;
221  points_tree.points_in_box(inpts, min, max);
222  for (size_type j = 0; j < inpts.size(); ++j)
223  if (retained_points.is_in(inpts[j].i) &&
224  constraints(inpts[j].i).contains(constraints(i))
225  && gmm::vect_dist2(points[i], points[inpts[j].i]) < d)
226  { kept = false; break; }
227  if (kept) {
228  if (noisy) cout << "kept point : " << points[i] << " co = "
229  << constraints(i) << endl;
230  retained_points.add(i);
231  }
232  }
233  }
234  }
235  }
236  return retained_points;
237  }

370  void mesh_level_set4::cut_element(size_type cv,
371  const dal::bit_vector &primary,
372  const dal::bit_vector &secondary,
373  scalar_type radius_cv) {
374 
375  cut_cv[cv] = convex_info();
376  cut_cv[cv].pmsh = pmesh(new mesh);
377  if (noisy) cout << "cutting element " << cv << endl;
378  bgeot::pgeometric_trans pgt = linked_mesh().trans_of_convex(cv);
379  std::auto_ptr<mesher_signed_distance> ref_element(new_ref_element(pgt));
380  std::vector<mesher_level_set> mesher_level_sets;
381 
382  size_type n = pgt->structure()->dim();
383  size_type nbtotls = primary.card() + secondary.card();
384  pintegration_method exactint = classical_exact_im(pgt);
385  GMM_ASSERT1(nbtotls <= 16,
386  "An element is cut by more than 16 level_set, aborting");
387 
388  /*
389  * Step 1 : build the signed distances, estimate the curvature radius
390  * and the degree K.
391  */
392  dim_type K = 0; // Max. degree of the level sets.
393  scalar_type r0 = 1E+10; // min curvature radius
394  std::vector<const mesher_signed_distance*> list_constraints;
395  point_stock mesh_points(list_constraints, radius_cv);
396 
397  ref_element->register_constraints(list_constraints);
398  size_type nbeltconstraints = list_constraints.size();
399  mesher_level_sets.reserve(nbtotls);
400  for (size_type ll = 0; ll < level_sets.size(); ++ll) {
401  if (primary[ll]) {
402  base_node X(n); gmm::fill_random(X);
403  K = std::max(K, (level_sets[ll])->degree());
404  mesher_level_sets.push_back(level_sets[ll]->mls_of_convex(cv, 0));
405  mesher_level_set &mls(mesher_level_sets.back());
406  list_constraints.push_back(&mesher_level_sets.back());
407  r0 = std::min(r0, curvature_radius_estimate(mls, X, true));
408  GMM_ASSERT1(gmm::abs(r0) >= 1e-13, "Something wrong in your level "
409  "set ... curvature radius = " << r0);
410  if (secondary[ll]) {
411  mesher_level_sets.push_back(level_sets[ll]->mls_of_convex(cv, 1));
412  mesher_level_set &mls2(mesher_level_sets.back());
413  list_constraints.push_back(&mesher_level_sets.back());
414  r0 = std::min(r0, curvature_radius_estimate(mls2, X, true));
415  }
416  }
417  }
418 
419  /*
420  * Step 2 : projection of points of a grid on each signed distance.
421  */
422  scalar_type h0 = std::min(1./(K+1), 0.2 * r0), dmin = 0.55;
423  bool h0_is_ok = true;
424 
425  do {
426 
427  h0_is_ok = true;
428  mesh_points.clear();
429  scalar_type geps = .001*h0;
430  size_type nbpt = 1;
431  std::vector<size_type> gridnx(n);
432  for (size_type i=0; i < n; ++i)
433  { gridnx[i]=1+(size_type)(1./h0); nbpt *= gridnx[i]; }
434  base_node P(n), Q(n), V(n);
435  dal::bit_vector co;
436  std::vector<size_type> co_v;
437 
438  for (size_type i=0; i < nbpt; ++i) {
439  for (size_type k=0, r = i; k < n; ++k) { // building grid point
440  unsigned p = unsigned(r % gridnx[k]);
441  P[k] = p * scalar_type(1) / scalar_type(gridnx[k]-1);
442  r /= gridnx[k];
443  }
444  co.clear(); co_v.resize(0);
445  if ((*ref_element)(P) < geps) {
446  bool kept = false;
447  for (size_type k=list_constraints.size()-1; k!=size_type(-1); --k) {
448  // for (size_type k=0; k < list_constraints.size(); ++k) {
449  // Rerversed to project on level set before on convex boundaries
450  gmm::copy(P, Q);
451  scalar_type d = list_constraints[k]->grad(P, V);
452  if (gmm::vect_norm2(V)*h0*7 > gmm::abs(d))
453  if (try_projection(*(list_constraints[k]), Q, true)) {
454  if (k >= nbeltconstraints
455  && gmm::vect_dist2(P, Q) < h0 * 3.5) kept = true;
456  if (gmm::vect_dist2(P, Q) < h0 / 1.5) {
457  co.add(k); co_v.push_back(k);
458  if ((*ref_element)(Q) < 1E-8) {
459  scalar_type r =
460  curvature_radius_estimate(*(list_constraints[k]), Q);
461  r0 = std::min(r0, r);
462  if (r0 < 4.*h0) { h0 = 0.2 * r0; h0_is_ok = false; break; }
463  if (k >= nbeltconstraints || kept) mesh_points.add(Q, r);
464  }
465  }
466  }
467  }
468  if (kept) mesh_points.add(P, 1.E10);
469  }
470  size_type nb_co = co.card();
471  dal::bit_vector nn;
472  if (h0_is_ok)
473  for (size_type count = 0; count < (size_type(1) << nb_co); ++count) {
474  nn.clear();
475  for (size_type j = 0; j < nb_co; ++j)
476  if (count & (size_type(1) << j)) nn.add(co_v[j]);
477  if (nn.card() > 1 && nn.card() <= n) {
478  if (noisy) cout << "trying set of constraints" << nn << endl;
479  gmm::copy(P, Q);
480  bool ok=pure_multi_constraint_projection(list_constraints,Q,nn);
481  if (ok && (*ref_element)(Q) < 1E-9) {
482  if (noisy) cout << "Intersection point found " << Q << " with "
483  << nn << endl;
484  mesh_points.add(Q);
485  }
486  }
487  }
488  }
489 
490  if (!h0_is_ok) continue;
491 
492  /*
493  * Step 3 : Delaunay, test if a simplex cut a level set and adapt
494  * the mesh to the curved level sets.
495  */
496 
497  dmin = 2.*h0;
498  if (noisy) cout << "dmin = " << dmin << " h0 = " << h0 << endl;
499  if (noisy) cout << "cetait le convexe " << cv << endl;
500 
501  dal::bit_vector retained_points
502  = mesh_points.decimate(*ref_element, dmin);
503 
504  bool delaunay_again = true;
505 
506  std::vector<base_node> fixed_points;
507  std::vector<dal::bit_vector> fixed_points_constraints;
508  mesh &msh(*(cut_cv[cv].pmsh));
509 
510  mesh_region &ls_border_faces(cut_cv[cv].ls_border_faces);
511  std::vector<base_node> cvpts;
512 
513  size_type nb_delaunay = 0;
514 
515  while (delaunay_again) {
516  if (++nb_delaunay > 15)
517  { h0_is_ok = false; h0 /= 2.0; dmin = 2.*h0; break; }
518 
519  fixed_points.resize(0);
520  fixed_points_constraints.resize(0);
521  // std::vector<base_node> fixed_points;
522  // std::vector<dal::bit_vector> fixed_points_constraints;
523  fixed_points.reserve(retained_points.card());
524  fixed_points_constraints.reserve(retained_points.card());
525  for (dal::bv_visitor i(retained_points); !i.finished(); ++i) {
526  fixed_points.push_back(mesh_points[i]);
527  fixed_points_constraints.push_back(mesh_points.constraints(i));
528  }
529 
530  gmm::dense_matrix<size_type> simplexes;
531  run_delaunay(fixed_points, simplexes, fixed_points_constraints);
532  delaunay_again = false;
533 
534  msh.clear();
535 
536  for (size_type i = 0; i < fixed_points.size(); ++i) {
537  size_type j = msh.add_point(fixed_points[i]); // remettre le add_norepeat ?
538  assert(j == i);
539  }
540 
541  cvpts.resize(0);
542  for (size_type i = 0; i < gmm::mat_ncols(simplexes); ++i) {
543  size_type j = msh.add_convex(bgeot::simplex_geotrans(n,1),
544  gmm::vect_const_begin(gmm::mat_col(simplexes, i)));
545  /* remove flat convexes => beware the final mesh may not be conformal ! */

937  int mesh_level_set4::is_not_crossed_by(size_type cv, plevel_set ls,
938  unsigned lsnum, scalar_type radius) {

940  GMM_ASSERT1(!mf.is_reduced(), "Internal error");

947  - different sign on the dof nodes => intersection for sure
948  - min value of the levelset greater than the radius of the convex
949  => no intersection
950  */

952  it != dofs.end(); ++it) {
953  scalar_type v = ls->values(lsnum)[*it];

960  base_node X(pf->dim()), G(pf->dim());

962  scalar_type d = mls1.grad(X, G);
963  if (gmm::vect_norm2(G)*2.5 < gmm::abs(d)) return p;

966  std::auto_ptr<mesher_signed_distance> ref_element(new_ref_element(pgt));
967 
968  gmm::fill_random(X); X *= 1E-2;
969  mesher_intersection mi1(*ref_element, mls1);
970  if (!try_projection(mi1, X)) return p;
971  if ((*ref_element)(X) > 1E-8) return p;

973  gmm::fill_random(X); X *= 1E-2;

975  mesher_intersection mi2(*ref_element, mls2);
976  if (!try_projection(mi2, X)) return p;
977  if ((*ref_element)(X) > 1E-8) return p;
978 
979  return 0;
