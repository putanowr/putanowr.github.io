122  void CircularSector::export_geo(std::ostream &out) const {
123  int nperim = GetNumOfSubsectors();
124  double start_angle = GetStartAngle();
125  double fill_angle = GetFillAngle();
126  double R = GetRadius();
127  double Ox,Oy;
128  bool internal_lines = HasInternalLines();
129  GetCenter(Ox,Oy);
130  GetParam("start_angle", start_angle);
131  std::vector<double> lc;
132  GetParam("lc", lc);
133  out << "Ox = " << Ox << ";\n";
134  out << "Oy = " << Oy << ";\n";
135  out << "R = " << R << ";\n";
136  out << "nperim = " << nperim << ";\n";
137  out << "fillangle = " << fill_angle << ";\n";
138  out << "dalpha = fillangle/nperim;\n";
139  out << "startangle = " << start_angle << ";\n";
140  out << "internal_lines = " << HasInternalLines() << ";\n";
141  out << "stp = newp;\n";
142  out << "Point(stp) = {Ox, Oy, 0, " << lc[0] << "}; // center\n";
143  double dalpha = fill_angle/nperim;
144  for (int i=0; i<GetNumOfVertices(); i++) {
145  double x = Ox + R*cos(start_angle + i*dalpha);
146  double y = Oy + R*sin(start_angle + i*dalpha);
147  out << "Point(newp) = {" << x << "," << y << ",0," << lc[i] << "};\n";
148  }
149  if (IsCircle()){
150  out << circle_template;
151  } else {
152  out << circular_sector_template;
153  }
154  }
