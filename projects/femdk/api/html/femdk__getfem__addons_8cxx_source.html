38  virtual void asm_real_tangent_terms(const model &md, size_type ib,
39  const model::varnamelist &vl,
40  const model::varnamelist &dl,
41  const model::mimlist &mims,
42  model::real_matlist &matl,
43  model::real_veclist &vecl,
44  model::real_veclist &,
45  size_type region,
46  build_version version) const {
47  GMM_ASSERT1(matl.size() == 1,
48  "Basic d/dt brick has one and only one term");
49  GMM_ASSERT1(mims.size() == 1,
50  "Basic d/dt brick need one and only one mesh_im");
51  GMM_ASSERT1(vl.size() == 1 && dl.size() >= 2 && dl.size() <= 3,
52  "Wrong number of variables for lumped d/dt brick");
53 
54  // It should me more convenient not to recompute the matrix if only
55  // dt is modified
56  bool recompute_matrix = !((version & model::BUILD_ON_DATA_CHANGE) != 0)
57  || (md.is_var_newer_than_brick(dl[1], ib));
58  if (dl.size() > 2)
59  recompute_matrix = recompute_matrix ||
60  md.is_var_newer_than_brick(dl[2], ib);
61 
62 
63  if (recompute_matrix) {
64  const mesh_fem &mf_u = md.mesh_fem_of_variable(vl[0]);
65  const mesh &m = mf_u.linked_mesh();
66  const mesh_im &mim = *mims[0];
67  mesh_region rg(region);
68  m.intersect_with_mpi_region(rg);
69 
70  const model_real_plain_vector &dt = md.real_variable(dl[1]);
71  GMM_ASSERT1(gmm::vect_size(dt) == 1, "Bad format for time step");
72 
73  const mesh_fem *mf_rho = 0;
74  const model_real_plain_vector *rho = 0;
75 
76  if (dl.size() > 2) {
77  mf_rho = md.pmesh_fem_of_variable(dl[2]);
78  rho = &(md.real_variable(dl[2]));
79  size_type sl = gmm::vect_size(*rho);
80  if (mf_rho) sl = sl * mf_rho->get_qdim() / mf_rho->nb_dof();
81  GMM_ASSERT1(sl == 1, "Bad format for density");
82  }
83 
84  GMM_TRACE2("Mass matrix assembly for d_on_dt brick");
85  if (dl.size() > 2 && mf_rho) {
86  gmm::clear(matl[0]);
87  asm_lumped_mass_matrix_param(matl[0], mim, mf_u, *mf_rho, *rho, rg);
88  gmm::scale(matl[0], scalar_type(1) / dt[0]);
89  } else {
90  gmm::clear(matl[0]);
91  asm_lumped_mass_matrix(matl[0], mim, mf_u, rg);
92  if (dl.size() > 2) gmm::scale(matl[0], (*rho)[0] / dt[0]);
93  else gmm::scale(matl[0], scalar_type(1) / dt[0]);
94  }
95  }
96  gmm::mult(matl[0], md.real_variable(dl[0], 1), vecl[0]);
97  }

171  (model &md, const mesh_im &mim, const std::string &varname,
172  const std::string &dataname_dt, const std::string &dataname_rho,
173  size_type region) {
