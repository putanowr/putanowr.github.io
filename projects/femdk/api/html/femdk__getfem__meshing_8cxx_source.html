52 void make_segment_mesh(getfem::mesh &mesh, const getfem::base_node &begin,
53  const getfem::base_node &end,
54  meshing_params param) {
55 
56  getfem::base_node dp = (end - begin)/((double)(param.degree*param.N));
57  getfem::base_node A = begin;
58 
59  /* Mesh for the segment */
60  getfem::mesh smesh;
61 
62  for (getfem::size_type i = 0; i<param.N; i++) {
63  std::vector<getfem::size_type> pi(param.degree+1);
64  for (getfem::size_type j=0; j<param.degree+1; ++j) {
65  getfem::base_node x = begin;
66  gmm::add(A, gmm::scaled(dp, (double)(i*param.degree+j)), x);
67  pi[j] = smesh.add_point(x);
68  }
69  smesh.add_convex(bgeot::simplex_geotrans(1, param.degree),
70  pi.begin());
71  }
72  if (param.overwrite) {
73  mesh.clear();
74  }
75  merge_meshes(mesh, smesh);
76 }

92 void make_single_tet_mesh(getfem::mesh &mesh, bool add_regions) {
93  std::vector<bgeot::base_node> pts(4);
94  pts[0] = bgeot::base_node(0.0, 0.0, 0.0);
95  pts[1] = bgeot::base_node(1.0, 0.0, 0.0);
96  pts[2] = bgeot::base_node(0.0, 1.0, 0.0);
97  pts[3] = bgeot::base_node(0.0, 0.0, 1.0);
98  getfem::size_type c;
99  c = mesh.add_convex_by_points(bgeot::simplex_geotrans(3,1), pts.begin());
100  if (true == add_regions) {
101  getfem::mesh_region X = mesh.region(1); // region normal to X
102  getfem::mesh_region Y = mesh.region(2); // region normal to Y
103  getfem::mesh_region Z = mesh.region(3); // region normal to Z
104  getfem::mesh_region XYZ = mesh.region(4); // the remaining face
105  X.add(c, 1);
106  Y.add(c, 2);
107  Z.add(c, 3);
108  XYZ.add(c,0);
109  }
110 }

203 int GMSHMesher::generate_kernel(const bgd::BGD &geom, meshing_params param,
204  const std::string &inputfile,
205  const std::string &outputfile) {
206  try {
207  geom.Export(inputfile.c_str(), "geo");
208  } catch (FEMDKException e) {
209  std::cerr << "The geometry: '' does not support gmsh export " << "\n";
210  std::cerr << e.what() << "\n";
211  exit(3);
212  }
213 
214  std::ofstream outfile;
215  outfile.open(inputfile.c_str(),std::ios_base::app);
216  switch(param.mesh_family) {
217  case SIMPLICAL_MESH:
218  break;
219  case CUBICAL_MESH:
220  outfile << "__nsur=news;\n";
221  outfile << "Recombine Surface{__nsur-1};\n";
222  outfile << "Mesh.SubdivisionAlgorithm=1;\n";
223  break;
224  case MIXED_MESH:
225  outfile << "__nsur=news;\n";
226  outfile << "Recombine Surface{__nsur-1};\n";
227  break;
228  default:
229  throw FEMDKException("Internal error, invalid mesh family type");
230  }
231  outfile.close();
232  std::vector<std::string> args;
233  args.push_back(getExecName());
234  int dim=param.target_dim;
235  if (0>dim || dim>3) {
236  dim = geom.GetDimension();
237  }
238 
239  std::ostringstream s;
240  if (false == param.verbose) {
241  args.push_back("-v");
242  args.push_back("0");
243  }
244  s << "-" << dim;
245  args.push_back(s.str());
246  args.push_back("-o");
247  args.push_back(outputfile);
248  args.push_back(inputfile);
249  return ExternalMesher::generate(getExecName(), args, NULL, NULL);
250 }

299 void AnnulusMapper::map(getfem::mesh &mesh) {
300  getfem::mesh::PT_TAB &pts = mesh.points();
301  for (getfem::size_type i = 0; i < pts.size(); ++i) {
302  double s = pts[i][0];
303  double t = pts[i][1];
304  double r = myR0*(1-s)+myR1*s;
305  getfem::base_node n(r*cos(myAlpha*t), r*sin(myAlpha*t));
306  pts[i] = n;
307  }
308  pts.resort();
309 }
