203  PrivContainer array;
204  array.push_back(PrivContainer::value_type("",
205  PrivContainer(boost::lexical_cast<std::string>(x))));
206  array.push_back(PrivContainer::value_type("",

210  myPoints.push_back(std::make_pair(label, array));
