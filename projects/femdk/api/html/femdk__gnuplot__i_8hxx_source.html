849 //
850 // A string tokenizer taken from http://www.sunsite.ualberta.ca/Documentation/
851 // /Gnu/libstdc++-2.90.8/html/21_strings/stringtok_std_h.txt
852 //
853 template <typename Container>
854 void stringtok (Container &container,
855  std::string const &in,
856  const char * const delimiters = " \t\n")
857 {
858  const std::string::size_type len = in.length();
859  std::string::size_type i = 0;
860 
861  while ( i < len )
862  {
863  // eat leading whitespace
864  i = in.find_first_not_of (delimiters, i);
865 
866  if (i == std::string::npos)
867  return; // nothing left but white space
868 
869  // find the end of the token
870  std::string::size_type j = in.find_first_of (delimiters, i);
871 
872  // push token
873  if (j == std::string::npos)
874  {
875  container.push_back (in.substr(i));
876  return;
877  }
878  else
879  container.push_back (in.substr(i, j-i));
880 
881  // set up for next loop
882  i = j + 1;
883  }
884 
885  return;
886 }
