32 bool string2double(const char *str, double *val) {
33  bool result = false;
34  char *endptr;
35  double v;
36  v = strtod(str, &endptr);
37  if (endptr == str) {
38  result = false;
39  } else {
40  if (NULL != val) {
41  *val = v;
42  result = true;
43  } else {
44  FEMDK_ERROR("Unexpected NULL argument");
45  }
46  }
47  return result;
48 }

98 void param2vector(const std::string &param_name, const std::string &input,
99  std::vector<double> &vec, const std::string &separator) {
100  // passing -1 as the submatch index parameter performs splitting
101  std::regex re(separator);
102  std::sregex_token_iterator
103  first{input.begin(), input.end(), re, -1},
104  last;
105  auto components = std::vector<std::string>{first, last};
106  for (auto &&str : components) {
107  double val;
108  param2double(param_name.c_str(), str.c_str(), val);
109  vec.push_back(val);
110  }
111 }
