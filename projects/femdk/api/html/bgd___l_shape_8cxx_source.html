40  void LShape::export_geo(std::ostream &out) const {
41  double b,Sb,g,H,Ox,Oy;
42  std::vector<double> lc(6,1.0);
43  GetParam("H", H);
44  GetParam("Sb", Sb);
45  GetParam("b", b);
46  GetParam("g", g);
47  GetParam("Ox", Ox);
48  GetParam("Oy", Oy);
49  GetParam("lc", lc);
50  out << "H = " << H << ";\n";
51  out << "Sb = " << Sb << ";\n";
52  out << "g = " << g << ";\n";
53  out << "b = " << b << ";\n";
54  out << "Ox = " << Ox << ";\n";
55  out << "Oy = " << Oy << ";\n";
56  out << "lc[] = {";
57  {
58  int i=0;
59  for (; i<lc.size()-1; i++) {
60  out << lc[i] << ",";
61  }
62  out << lc[i] << "};\n";
63  }
64 
65  out << "stp = newp;\n"
66  "Point(stp) = {Ox, Oy, 0, lc[0]};\n"
67  "Point(newp) = {Ox+Sb,Oy, 0, lc[1]};\n"
68  "Point(newp) = {Ox+Sb,Oy+b, 0, lc[2]};\n"
69  "Point(newp) = {Ox+g, Oy+b, 0, lc[3]};\n"
70  "Point(newp) = {Ox+g, Oy+H, 0, lc[4]};\n"
71  "Point(newp) = {Ox, Oy+H, 0, lc[5]};\n";
72  out << "\n";
73  out << "outl[] = {};\n";
74  out << "For i In {0:5}\n"
75  " __ln = newl;\n"
76  " Line(__ln) = {stp+i, stp+Fmod(i+1,6)};\n"
77  " outl[] += __ln;\n"
78  "EndFor\n"
79  "\n";
80  out << "perim = newll;\n"
81  "Line Loop(perim) = {outl[]};\n"
82  "\n"
83  " sur = news;\n"
84  "Plane Surface(sur) = {perim};\n"
85  "\n";
86  for (int i=0; i<6;i++) {
87  out << "Physical Line(\"b" << i+1 << "\")= {outl[" << i << "]};\n";
88  }
89  out << "Physical Surface(\"all\") = {sur};\n";
90  }
