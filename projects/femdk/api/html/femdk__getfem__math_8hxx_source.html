49 template <class V>
50 void linspace(const V &p1, const V &p2, int N, std::vector<V> &pts) {
51  V dp = p2-p1;
52  V pcurr(dp.size());
53  if (N<2) {
54  N=2;
55  }
56  double dt=1.0/(N-1);
57  for (int i=0;i<N; i++) {
58  gmm::add(p1, gmm::scaled(dp, i*dt), pcurr);
59  pts.push_back(pcurr);
60  }
61 }

185 inline void
186 SegmentInterpolator::find_lambda(const bgeot::base_node &a,
187  const bgeot::base_node &b,
188  const bgeot::base_node &x,
189  bgeot::base_node &lambda) const {
190  double xx = std::accumulate(x.begin(), x.end(), 0.0);
191  double xp = std::accumulate(a.begin(), a.end(), 0.0);
192  double xk = std::accumulate(b.begin(), b.end(), 0.0);
193  if (fabs(xp - xk) < 1.e-12) {
194  lambda[0] = 0.5;
195  } else {
196  lambda[0] = (xx - xp)/(xk - xp);
197  }
198  if (lambda[0] < 0.0) {
199  lambda[0] = 0.0;
200  } else if (lambda[0] > 1.0) {
201  lambda[0] = 1.0;
202  }
203 }

216 SegmentInterpolator::EvalAt(const double lambda) const {
217  assert(NULL != myInterpolator);
218  bgeot::base_node p(1, lambda);
219  if (p[0] < 0) { p[0] = 0.0;}
220  if (p[0] > 1) { p[0] = 1.0;}
221  bool result = myInterpolator->eval(p,myVal, myGrad);
222  assert(result == true);
223  return myVal[0];
