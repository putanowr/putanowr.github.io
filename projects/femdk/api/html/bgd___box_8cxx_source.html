66  out << "Nz = " << Nz << ";\n";

68  "Point(stp) = {Ox, Oy, Oz, 1}; // 1 \n"

122  void Box::export_oogl(std::ostream &out, const ConfigData *options) const {
123  /* set default format */
124  std::string format = std::string("off");
125  if (nullptr != options) {
126  try {
127  format = options->get<std::string>("format");
128  } catch (boost::property_tree::ptree_bad_data e) {
129  FEMDK_ERROR(e.what());
130  } catch (boost::property_tree::ptree_bad_path e) {
131  /* use the default format */
132  }
133  }
134  if (format == "bbox") {
135  export_oogl_bbox(out);
136  } else if (format == "off") {
137  export_oogl_off(out);
138  } else {
139  FEMDK_ERROR("Invalid format for oogl output");
140  }
141  }

159  BGDErrorCode Box::Export(std::ostream &outfile,
160  const char *file_type,
161  const ConfigData *options) const {
162  if ( 0 == strcmp(file_type, "geo")) {
163  export_geo(outfile, options);
164  } else if ( 0 == strcmp(file_type, "vtk")) {
165  throw FEMDKNotImplemented();
166  } else if (0 == strcmp(file_type, "oogl")) {
167  export_oogl(outfile, options);
168  } else if (0 == strcmp(file_type, "tochnog")) {
169  export_tochnog(outfile, options);
170  } else {
171  std::ostringstream s;
172  s << "File format " << file_type << "not supported";
173  throw FEMDKException(s.str());
174  }
175  return BGD_SUCCESS;
176  }
