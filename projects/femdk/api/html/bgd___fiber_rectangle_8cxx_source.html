40  void FiberRectangle::export_geo(std::ostream &out) const {
41  double Lx,Ly,Ox,Oy;
42  double Fr, Fdx, Fdy, Fox, Foy, Flc;
43  int Nx,Ny;
44  int FNx, FNy;
45  std::vector<double> lc(4,1.0);
46  GetParam("Lx", Lx);
47  GetParam("Ly", Ly);
48  GetParam("Nx", Nx);
49  GetParam("Ny", Ny);
50  GetParam("Ox", Ox);
51  GetParam("Oy", Oy);
52  GetParam("lc", lc);
53  GetParam("Flc", Flc);
54  GetParam("FNx", FNx);
55  GetParam("FNy", FNy);
56  GetParam("Fdx", Fdx);
57  GetParam("Fdy", Fdy);
58  GetParam("Fox", Fox);
59  GetParam("Foy", Foy);
60  GetParam("Fr", Fr);
61  double ex = Lx/Nx;
62  double ey = Ly/Ny;
63  double e = ex < ey ? ex : ey;
64  for (int i=0; i<lc.size(); i++) {
65  if (lc[i] < 0) lc[i] = e;
66  }
67  for (int i=0; i<4-lc.size(); i++) {
68  lc.push_back(e);
69  }
70  out << "Lx = " << Lx << ";\n";
71  out << "Ly = " << Ly << ";\n";
72  out << "Ox = " << Ox << ";\n";
73  out << "Oy = " << Oy << ";\n";
74  out << "Fr = " << Fr << ";\n";
75  out << "Fdx = " << Fdx << ";\n";
76  out << "Fdy = " << Fdy << ";\n";
77  out << "Fox = " << Fox << ";\n";
78  out << "Foy = " << Foy << ";\n";
79  out << "Flc = " << Flc << ";\n";
80  out << "stp = newp;\n";
81  out << "lc = 1;\n"
82  "Point(stp) = {Ox, Oy, 0," << lc[0] << "}; // 1\n"
83  "Point(newp) = {Ox+Lx,Oy, 0," << lc[1] << "}; // 2\n"
84  "Point(newp) = {Ox+Lx,Oy+Ly,0," << lc[2] << "}; // 3\n"
85  "Point(newp) = {Ox, Oy+Ly,0," << lc[3] << "}; // 4\n"
86  "\n"
87  "outl[] = {};\n"
88  "southLin[] = {};\n"
89  "eastLin[]= {};\n"
90  "northLin[] = {};\n"
91  "westLin[] = {};\n"
92  "\n"
93  "__ln = newl;\n"
94  "Line(__ln) = {stp+0, stp+1};\n"
95  "southLin[] += __ln;\n"
96  "outl[] += __ln;\n"
97  "\n"
98  "__ln = newl;\n"
99  "Line(__ln) = {stp+1, stp+2};\n"
100  "eastLin[] += __ln;\n"
101  "outl[] += __ln;\n"
102  "\n"
103  "__ln = newl;\n"
104  "Line(__ln) = {stp+2, stp+3};\n"
105  "northLin[] += __ln;\n"
106  "outl[] += __ln;\n"
107  "\n"
108  "__ln = newl;\n"
109  "Line(__ln) = {stp+3, stp+0};\n"
110  "westLin[] += __ln;\n"
111  "outl[] += __ln;\n"
112  "\n"
113  "perim = newll;\n"
114  "Line Loop(perim) = {outl[]};\n"
115  "\n"
116  " sur = news;\n"
117  "Plane Surface(sur) = {perim};\n"
118  "\n"
119  "Physical Line(\"south\")= {southLin[]};\n"
120  "Physical Line(\"east\")= {eastLin[]};\n"
121  "Physical Line(\"north\")= {northLin[]};\n"
122  "Physical Line(\"west\")= {westLin[]};\n"
123  "Physical Surface(\"all\") = {sur};\n";
124 
125  out << "For i In{0, FNx}\n";
126  out << " For j In{0, FNy}\n";
127  out << " __fcp = newp;\n";
128  out << " __fcx = Ox+Fox+i*Fdx;\n";
129  out << " __fcy = Oy+Foy+j*Fdy;\n";
130  out << " Point(fcp) = {__fcx,__fcy ,0, Flc};\n";
131  out << " Point(newp) = {__fcx+Fr,__fcy ,0, Flc};\n";
132  out << " Point(newp) = {__fcx,__fcy+Fr ,0, Flc};\n";
133  out << " Point(newp) = {__fcx-Fr,__fcy ,0, Flc};\n";
134  out << " Point(newp) = {__fcx,__fcy-Fr ,0, Flc};\n";
135  out << " For k In{0,4}\n;";
136  out << " __ln = newline;";
137  out << " Circle(__ln) = {__fcp+Fmod(k+1,4),__fcp,"
138  "__fcp+Fmod(k+1,4)};\n";
139  out << " EndFor\n";
140  out << "EndFor\n";
141  }
