37 void setConfigureRegistry(void) {

40  called = true;

45  }

91 bool configure_Rectangle(BGD *shape, const ConfigData &shapeConfig) {
92  const char *doubleParams[] = {"Lx", "Ly", "Ox", "Oy"};
93  size_t nd = sizeof(doubleParams)/sizeof(char*);

96  const char *intParams[] = {"Nx", "Ny"};

98  configure_int_params(shape, shapeConfig, intParams, ni);
99 
100  std::vector<double> lc;
